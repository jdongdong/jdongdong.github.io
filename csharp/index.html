常用验证
using System.Text.RegularExpressions;
/// <summary>
/// 验证手机号
/// </summary>
/// <param name="s"></param>
/// <returns></returns>
public static bool IsMobile(string s)
{
    return Regex.IsMatch(s, @"^1[3578]\d{9}$");
}
/// <summary>
/// 验证邮箱
/// </summary>
/// <param name="s"></param>
/// <returns></returns>
public static bool IsEmail(string s)
{
    String strExp = @"\w+([-+.]\w+)*@\w+([-.]\w+)*\.\w+([-.]\w+)*";
    Regex r = new Regex(strExp);
    Match m = r.Match(s);
    return m.Success;
}
/// <summary>
/// 验证电话号码
/// </summary>
/// <param name="s"></param>
/// <returns></returns>
public static bool IsTelphone(string s)
{
    return Regex.IsMatch(s, @"^(\d{3,4}-)?\d{6,8}$");
}
/// <summary>
/// 验证字母或数字
/// </summary>
/// <param name="s"></param>
/// <returns></returns>
public static bool IsLetterOrNumber(string s)
{
    return Regex.IsMatch(s, "^[A-Za-z0-9]*$");
}
